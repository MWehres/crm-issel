FIRESTORE SERVICE: Kalkulationen verwalten ──────────────────────────────────────────────────────────── */ class KalkulationService { /** * Neue Kalkulation in Firestore speichern * @param {string} dealId - ID des Deals (Contact) * @param {object} kalkulationData - Kalkulationsdaten * @returns {Promise} - ID der gespeicherten Kalkulation */ static async create(dealId, kalkulationData) { try { // Version ermitteln (automatisch hochzählen) const version = await this.getNextVersion(dealId); // Timestamp hinzufügen + User-Info const dataToSave = { ...kalkulationData, version: version, createdAt: firebase.firestore.FieldValue.serverTimestamp(), createdBy: firebase.auth().currentUser?.uid || 'anonymous' }; // In Firestore speichern const docRef = await firebase.firestore() .collection('contacts') // ← Deine Contact-Collection .doc(dealId) .collection('kalkulationen') // ← Subcollection für Kalkulationen .add(dataToSave); console.log('✓ Kalkulation gespeichert:', docRef.id); // Deal-Value aktualisieren wenn Gesamtsumme vorhanden if (kalkulationData.gesamtsumme) { await this.updateContactValue(dealId, kalkulationData.gesamtsumme); } return docRef.id; } catch (error) { console.error('❌ Fehler beim Speichern:', error); throw error; } } /** * Alle Kalkulationen für einen Deal laden * @param {string} dealId - Deal-ID * @returns {Promise} - Array mit Kalkulationen */ static async getAll(dealId) { try { const snapshot = await firebase.firestore() .collection('contacts') .doc(dealId) .collection('kalkulationen') .orderBy('createdAt', 'desc') // Neueste zuerst .get(); return snapshot.docs.map(doc => ({ id: doc.id, ...doc.data(), // Firestore Timestamp → JavaScript Date createdAt: doc.data().createdAt?.toDate() })); } catch (error) { console.error('❌ Fehler beim Laden:', error); return []; } } /** * Einzelne Kalkulation laden * @param {string} dealId - Deal-ID * @param {string} kalkId - Kalkulationen-ID * @returns {Promise} - Kalkulation oder null */ static async getById(dealId, kalkId) { try { const doc = await firebase.firestore() .collection('contacts') .doc(dealId) .collection('kalkulationen') .doc(kalkId) .get(); if (!doc.exists) { console.warn('⚠️ Kalkulation nicht gefunden:', kalkId); return null; } return { id: doc.id, ...doc.data(), createdAt: doc.data().createdAt?.toDate() }; } catch (error) { console.error('❌ Fehler beim Laden:', error); return null; } } /** * Contact-Value (Deal-Wert) aktualisieren * @param {string} dealId - Deal-ID * @param {number} newValue - Neuer Wert */ static async updateContactValue(dealId, newValue) { try { await firebase.firestore() .collection('contacts') .doc(dealId) .update({ value: Math.round(newValue), lastCalculated: firebase.firestore.FieldValue.serverTimestamp() }); console.log('✓ Deal-Value aktualisiert:', newValue, '€'); } catch (error) { console.error('❌ Fehler beim Deal-Update:', error); // Nicht kritisch - weitermachen } } /** * Nächste Versionsnummer ermitteln * @param {string} dealId - Deal-ID * @returns {Promise} - Nächste Version */ static async getNextVersion(dealId) { const kalks = await this.getAll(dealId); if (kalks.length === 0) return 1; const versions = kalks.map(k => k.version || 0); return Math.max(...versions) + 1; } /** * Kalkulation löschen * @param {string} dealId - Deal-ID * @param {string} kalkId - Kalkulationen-ID */ static async delete(dealId, kalkId) { if (!confirm('Kalkulation wirklich löschen?')) return; try { await firebase.firestore() .collection('contacts') .doc(dealId) .collection('kalkulationen') .doc(kalkId) .delete(); console.log('✓ Kalkulation gelöscht:', kalkId); return true; } catch (error) { console.error('❌ Fehler beim Löschen:', error); alert('Fehler beim Löschen: ' + error.message); return false; } } /** * Real-Time Updates für Kalkulationen * @param {string} dealId - Deal-ID * @param {function} callback - Callback mit Kalkulationen-Array * @returns {function} - Unsubscribe-Funktion */ static listenToUpdates(dealId, callback) { return firebase.firestore() .collection('contacts') .doc(dealId) .collection('kalkulationen') .orderBy('createdAt', 'desc') .onSnapshot( snapshot => { const kalkulationen = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data(), createdAt: doc.data().createdAt?.toDate() })); callback(kalkulationen); }, error => { console.error('❌ Snapshot-Fehler:', error); } ); } } /* ──────────────────────────────────────────────────────────── 2. /** * Kalkulations-Modal öffnen * @param {string} dealId - Contact/Deal-ID */ function openKalkulation(dealId) { currentDealId = dealId; // Contact-Daten aus deinem bestehenden Array laden // ANPASSEN: Je nachdem wo deine Kontakte gespeichert sind const contact = contacts.find(c => c.id === dealId); if (!contact) { alert('Kontakt nicht gefunden'); return; } currentContactData = contact; // Modal öffnen const modal = document.getElementById('kalk-modal'); modal.classList.add('active'); document.body.style.overflow = 'hidden'; // Scrollen verhindern // Iframe-Referenz speichern kalkulationIframe = document.getElementById('kalk-iframe'); // Kalkulationstool laden // ANPASSEN: Pfad zu deinem Kalkulationstool kalkulationIframe.src = 'SVS-Kalkulationstool_ISSSeL_v3_36.html'; // Deal-Info im Header anzeigen const dealInfo = document.getElementById('kalk-deal-info'); dealInfo.textContent = `${contact.company} · ${contact.firstname || ''} ${contact.lastname || ''} · ${formatCurrency(contact.value || 0)}`; // Warten bis Iframe geladen ist kalkulationIframe.onload = () => { console.log('✓ Kalkulationstool geladen'); // Kurze Verzögerung damit iframe-Skripte initialisiert sind setTimeout(() => { syncDataToKalkulation(); }, 800); }; // Kalkulationen-Historie laden loadKalkulationHistory(dealId); } /** * Kontaktdaten ins Kalkulationstool übertragen */ function syncDataToKalkulation() { if (!kalkulationIframe || !currentContactData) { console.warn('⚠️ Sync nicht möglich - Iframe oder Daten fehlen'); return; } try { const iframeWindow = kalkulationIframe.contentWindow; const iframeDoc = iframeWindow.document; // Mapping: CRM-Felder → Kalkulationstool-Felder const fieldMapping = { 'f-kunde': currentContactData.company || '', 'f-ansp': `${currentContactData.firstname || ''} ${currentContactData.lastname || ''}`.trim(), 'f-strasse': currentContactData.street || '', 'f-plz': currentContactData.zip || '', 'f-ort': currentContactData.city || '', 'f-email': currentContactData.email || '', 'f-telefon': currentContactData.phone || currentContactData.mobile || '', 'f-anrede': 'Herr' // Standard, ggf. aus Kontakt }; // Felder befüllen let filledCount = 0; Object.entries(fieldMapping).forEach(([fieldId, value]) => { const field = iframeDoc.getElementById(fieldId); if (field && value) { field.value = value; // Events triggern damit das Kalkulationstool reagiert field.dispatchEvent(new Event('input', { bubbles: true })); field.dispatchEvent(new Event('change', { bubbles: true })); filledCount++; } }); // Datum setzen (heute) const datumField = iframeDoc.getElementById('f-datum'); if (datumField && !datumField.value) { datumField.value = new Date().toISOString().split('T')[0]; } console.log(`✓ ${filledCount} Felder synchronisiert`); // Visuelles Feedback showSyncNotification('✓ Kontaktdaten übernommen'); } catch (error) { console.error('❌ Sync-Fehler:', error); showSyncNotification('⚠️ Sync-Fehler - Bitte manuell prüfen', true); } } /** * Kalkulation in Firestore speichern */ async function saveKalkulation() { if (!kalkulationIframe || !currentDealId) { alert('Fehler: Kein Deal ausgewählt'); return; } try { const iframeWindow = kalkulationIframe.contentWindow; // Validierung if (!kalkulationData.kunde) { alert('Bitte Kundennamen im Kalkulationstool eintragen'); return; } // In Firestore speichern showSyncNotification('💾 Speichere Kalkulation...'); const kalkId = await KalkulationService.create(currentDealId, kalkulationData); console.log('✓ Kalkulation gespeichert:', kalkId); showSyncNotification('✓ Kalkulation gespeichert!'); // Modal nach 1.5 Sekunden schließen setTimeout(() => { closeKalkulation(); // OPTIONAL: Contact-Liste neu laden // loadContacts(); // Deine bestehende Funktion }, 1500); } catch (error) { console.error('❌ Speichern fehlgeschlagen:', error); alert('Fehler beim Speichern:\n' + error.message); } } /** * Kalkulation aus Historie laden * @param {string} kalkId - Kalkulationen-ID */ async function loadKalkulationFromHistory(kalkId) { if (!currentDealId || !kalkulationIframe) return; try { showSyncNotification('📂 Lade Kalkulation...'); const kalk = await KalkulationService.getById(currentDealId, kalkId); if (!kalk || !kalk.snapshot) { alert('Kalkulation konnte nicht geladen werden'); return; } // Snapshot ins Kalkulationstool laden const iframeWindow = kalkulationIframe.contentWindow; if (iframeWindow.restoreSnapshot) { iframeWindow.restoreSnapshot(kalk.snapshot); showSyncNotification('✓ Kalkulation geladen: ' + kalk.name); // Historie-Panel schließen const historyPanel = document.getElementById('kalk-history-panel'); if (historyPanel) { historyPanel.style.display = 'none'; } } else { alert('Kalkulationstool unterstützt kein Laden (restoreSnapshot fehlt)'); } } catch (error) { console.error('❌ Laden fehlgeschlagen:', error); alert('Fehler beim Laden:\n' + error.message); } } /** * Kalkulationen-Historie laden und anzeigen * @param {string} dealId - Deal-ID */ async function loadKalkulationHistory(dealId) { const historyPanel = document.getElementById('kalk-history-panel'); const historyList = document.getElementById('kalk-history-list'); const historyCount = document.getElementById('kalk-history-count'); if (!historyPanel || !historyList) return; try { const kalkulationen = await KalkulationService.getAll(dealId); // Anzahl aktualisieren if (historyCount) { historyCount.textContent = kalkulationen.length; } // Liste befüllen if (kalkulationen.length === 0) { historyList.innerHTML = `
Noch keine Kalkulationen vorhanden
`; return; } historyList.innerHTML = kalkulationen.map(k => `
📄
${escapeHtml(k.name || 'Angebot')}
📅 ${k.createdAt ? k.createdAt.toLocaleDateString('de-DE') : '—'} 🔢 v${k.version || 1}
${formatCurrency(k.gesamtsumme || 0)}
`).join(''); } catch (error) { console.error('❌ Historie laden fehlgeschlagen:', error); historyList.innerHTML = `
Fehler beim Laden der Historie
`; } } /** * Historie-Panel toggle */ function toggleKalkulationHistory() { const panel = document.getElementById('kalk-history-panel'); if (!panel) return; const isVisible = panel.style.display !== 'none'; panel.style.display = isVisible ? 'none' : 'block'; } /** * Modal schließen */ function closeKalkulation() { const modal = document.getElementById('kalk-modal'); modal.classList.remove('active'); document.body.style.overflow = ''; // Scrollen wieder erlauben // Cleanup nach Animation setTimeout(() => { if (kalkulationIframe) { kalkulationIframe.src = ''; // Iframe entladen } currentDealId = null; currentContactData = null; kalkulationIframe = null; }, 300); } /* ──────────────────────────────────────────────────────────── 3. HILFSFUNKTIONEN ──────────────────────────────────────────────────────────── */ /** * Wert aus Kalkulationstool-Element extrahieren * @param {Document} doc - Iframe-Document * @param {string} elementId - Element-ID * @returns {number} - Extrahierter Wert */ function extractCalcValue(doc, elementId) { const el = doc.getElementById(elementId); if (!el) return 0; const text = el.textContent || el.value || '0'; const cleaned = text.replace(/[^\d,.-]/g, '').replace(',', '.'); return parseFloat(cleaned) || 0; } /** * Währung formatieren * @param {number} value - Betrag * @returns {string} - Formatierter String */ function formatCurrency(value) { return new Intl.NumberFormat('de-DE', { style: 'currency', currency: 'EUR', minimumFractionDigits: 0, maximumFractionDigits: 0 }).format(value); } /** * HTML escapen (XSS-Schutz) * @param {string} text - Text * @returns {string} - Escaped Text */ function escapeHtml(text) { const div = document.createElement('div'); div.textContent = text; return div.innerHTML; } /** * Sync-Notification anzeigen * @param {string} message - Nachricht * @param {boolean} isError - Fehler? */ function showSyncNotification(message, isError = false) { const notification = document.getElementById('kalk-sync-notification'); if (!notification) return; notification.textContent = message; notification.style.background = isError ? 'var(--red)' : 'var(--green)'; notification.classList.add('visible'); setTimeout(() => { notification.classList.remove('visible'); }, 3000); } /* ──────────────────────────────────────────────────────────── 4.
